Свой плеер — кнопка Play/Pause через JS
addEventListener, методы play() и pause(), свойство paused
Заблокировано ▶
Задание
Уберите у тега <video> атрибут controls и сделайте свою кнопку «Play». При нажатии на кнопку видео должно запускаться, при повторном нажатии — ставиться на паузу. Текст кнопки также меняется: «Play» → «Pause» → «Play». Используйте обработчик события click и проверку свойства paused.
Превью — нажми Play
кнопка переключает воспроизведение
Подсказка — HTML
<video width="480" id="myVideo">
  <source src="video.mp4" type="video/mp4">
</video>
<button id="playBtn">Play</button>
Подсказка — JavaScript
var video = document.getElementById('myVideo');
var btn   = document.getElementById('playBtn');

btn.addEventListener('click', function () {
  if (video.paused) {
    video.play();
  } else {
    video.pause();
  }
});

video.addEventListener('play', function () {
  btn.innerText = 'Pause';
});

video.addEventListener('pause', function () {
  btn.innerText = 'Play';
});
Логика: событие play срабатывает, когда видео началось, событие pause — когда остановилось. Меняем текст кнопки именно на этих событиях, а не сразу после клика — так надёжнее.

Громкость и таймер времени
input range, события timeupdate и canplaythrough
Заблокировано ▶
Задание
Добавьте к плееру из шага 3 регулятор громкости (<input type="range"> со значениями от 0 до 1, шагом 0.1) и текстовый таймер, показывающий текущее время воспроизведения в формате ММ:СС. Громкость задаётся через свойство video.volume, текущее время — через video.currentTime, а событие timeupdate срабатывает несколько раз в секунду пока видео идёт.
Превью
00:00
Подсказка — HTML
<video id="myVideo"> ... </video>
<button id="playBtn">Play</button>
<span id="timer">00:00</span>
<input type="range" id="volume"
       min="0" max="1" step="0.1" value="0.5">
Подсказка — JavaScript
var volume = document.getElementById('volume');
var timer  = document.getElementById('timer');

// меняем громкость при движении ползунка
volume.addEventListener('input', function () {
  video.volume = volume.value;
});

// обновляем таймер несколько раз в секунду
video.addEventListener('timeupdate', function () {
  var t = Math.floor(video.currentTime);
  var m = Math.floor(t / 60);
  var s = t % 60;
  if (s < 10) s = '0' + s;
  if (m < 10) m = '0' + m;
  timer.innerText = m + ':' + s;
});
Подробнее про события: timeupdate — срабатывает 4–10 раз в секунду пока идёт воспроизведение. canplaythrough — срабатывает один раз, когда браузер уверен, что сможет проиграть видео без задержек (на нём удобно показывать панель управления, изначально скрытую). ended — срабатывает когда видео закончилось.
